[docs]class Analysis(object):
    """Load historical data and analyse asset performance.

    Optionally set new to the name of a predefined asset to define
    the asset on initialisation.

    Args:
        new (string): Name of the new asset, which must already excisit in
        the database. Providing this name calls set_asset.

    Examples:
            Create and set an asset as amp and send it for analysis:

            >>> import flexibility as fl
            >>> amp = fl.Asset('amp')
            >>> a = Analysis(asset=amp)

            Or do this in one operation:

            >>> a = fl.Analysis(asset_name='amp')
    """
    def __init__(self, **kwargs):
        self._set_asset(**kwargs)
        self.schedules = Scheduling(self)
        self.contract = Contract(self)


    def _set_asset(self, asset=None, asset_name=None, refresh=False, cut=True):
        """Set the asset to be analysed.

        Use either `asset` or `asset_name` to define the asset. If both are
        erroniously defined `asset` takes priority.

        Args:
            asset (object): Asset as defined by asset.
            asset_name (str): Name of the asset
            refresh (bool): If true, loads data from data base after setting
                            asset. Default is `False`, where a serialised pickle
                            file is loaded.
            cut (bool): 
        """
        if asset:
            self.asset = asset
        elif asset_name:
            self.asset = Asset(asset_name)


        if not refresh:
            self._load_data(cut=cut)
        elif refresh:
            self._load_data(refresh=True, cut=cut)
            self._load_data(cut=cut)

    def _load_data(self, refresh=False, cut=True):
        """Load data from combined data sources.

        Data scources are described and combined in data subclass.
        Must have an asset defined to retrieve the data, as it is location
        dependent.
        """
        if not self.asset:
            raise KeyError('Asset not defined. Define asset before analysis')
        self.data = Data(self.asset, refresh,cut)



[docs]    def financial_summary(self, detailed=False, no_print=False):
        """Print key figures and findings:

        Args:
            detailed (bool): Default False, True not implimented yet
        """
        # Financial figures:
        # Payments to customer
        self.payments_to_customer = self.data.data['payment'].groupby(pd.TimeGrouper('M')).sum()
        s = self.payments_to_customer['imbalance'] + self.payments_to_customer['deviation']
        p = 21*1.77*24*np.array(self.payments_to_customer.index.day)
        self.payments_to_customer['Monthly'] = ((s>p)*s + (s<=p)*p )*(s.cumsum() <= p.cumsum()) + (s>p)*s*(s.cumsum() > p.cumsum())
        self.payments_to_customer['Excess'] = self.payments_to_customer['Monthly'] - p
        # add groupby as per: https://stackoverflow.com/questions/30017590/limit-cumsum-to-only-the-previous-4-rows
        # for YTD implimentation

        # HH
        self.financals ={}
        self.financals['returns_cov'] =  np.array(self.data.data[['profit']].cov())
        self.financals['returns_std'] = np.array(self.data.data[['profit']].std())
        self.financals['returns_mean'] = np.array(self.data.data[['profit']].mean())
        self.financals['sharpe_ratio'] = self.financals['returns_mean']/self.financals['returns_std']

        # Yearly
        # Customer
        self.financals['yearly_pay_to_amp'] = self.payments_to_customer['Excess'].sum()/len(self.payments_to_customer)*12
        self.financals['yearly_imba_dev_to_amp'] = self.payments_to_customer['Monthly'].sum()/len(self.payments_to_customer)*12
        self.financals['yearly_da_to_amp'] = self.payments_to_customer['da'].sum()/len(self.payments_to_customer)*12

        # DONG
        self.data.data['profit_total'] = self.data.data['profit_da'] + self.data.data['profit_wd'] + self.data.data['profit_imba']
        self.financals['DONG_profit_da'] = self.data.data['profit_da'].sum()/((self.data.data.index[-1] - self.data.data.index[0])/pd.Timedelta('365day'))
        self.financals['DONG_profit_wd'] = self.data.data['profit_wd'].sum()/((self.data.data.index[-1] - self.data.data.index[0])/pd.Timedelta('365day'))
        self.financals['DONG_profit_imbalance'] = self.data.data['profit_imba'].sum()/((self.data.data.index[-1] - self.data.data.index[0]) /pd.Timedelta('365day'))
        self.financals['DONG_profit_energy'] =  self.data.data['profit_total'].sum()/((self.data.data.index[-1] - self.data.data.index[0]) /pd.Timedelta('365day'))
        self.financals['DONG_profit_total'] = self.financals['DONG_profit_energy'] - self.financals['yearly_pay_to_amp']*(self.financals['yearly_pay_to_amp']<0)
        self.financals['DONG_profit_std'] = self.data.data['profit_total'].std()
        self.financals['DONG_profit_sharpe'] = self.data.data['profit_total'].mean()/self.data.data['profit_total'].std()

        if not no_print:
            print('Mean half hourly profits: DA:{:.2f}, WD:{:.2f}, Imbalance: {:.2f},'.format(self.financals['returns_mean'][0],
                                                                                              self.financals['returns_mean'][1],
                                                                                              self.financals['returns_mean'][2]))
            print('Sharpe ratio of HH returns: DA:{:.3}, WD:{:.3}, Imbalance:{:.3}'.format(self.financals['sharpe_ratio'][0],
                                                                                           self.financals['sharpe_ratio'][1],
                                                                                           self.financals['sharpe_ratio'][2]))
            print('\n')
            print('On a yearly basis the asset earned DONG:\n {:.2f} (Day ahead)\n+{:.2f} (within day)\n+{:.2f} (imbalance)\n-{:.2f} (payement)\n= {:.2f} (total)'.format(
                                                                                                                    self.financals['yearly_pay_to_amp']*(self.financals['yearly_pay_to_amp']<0),
                                                                                                                                            self.financals['DONG_profit_total']))
            if self.financals['yearly_pay_to_amp']<0:
                print("Payments to amp are {:.2f} in arrears.".format(-self.financals['yearly_pay_to_amp']))
            else:
                print("Payments to amp are {:.2f} in excess.".format(self.financals['yearly_pay_to_amp']))


            print('On a yearly basis the asset earned the customer: {:.2f} + {:.2f} = {:.2f}'.format(self.financals['yearly_imba_dev_to_amp'],
                                                                           self.financals['yearly_da_to_amp'],
                                                                            self.financals['yearly_da_to_amp']
                                                                            +self.financals['yearly_imba_dev_to_amp']))

        if detailed:
            # Convert GBP/half hour to kGBP/MW/a
            factor = 365*48/self.asset.power*(
                            (self.data.data.index[-1]-self.data.data.index[0])
                            /pd.Timedelta('1day'))/1000
            ylabel = 'Profit kGBP/MW/a'
            #ylabel = 'Mean half hourly profits GBP'

            plt.figure(figsize=(3, 4))
            sns.barplot( data=self.data.data['profit']*factor, capsize =0.1)
            plt.ylim(0, 25)
            plt.ylabel(ylabel)
            plt.xlabel('Market')
            plt.show()




[docs]    def trading_summary(self, detailed=False, no_print=False):
        """Display a summary of trading activity

        Args:
            detailed (bool): Default False, True witll plot some a distribution of the workload
        """
        self.trader_workload = {}
        self.trader_workload['max_trades_per_HH'] = self.data.data['info_offers_count'].max()
        self.trader_workload['95_trades_per_HH'] = self.data.data['info_offers_count'].quantile(0.95)
        self.trader_workload['max_trades_per_h'] = self.data.data['info_offers_count'].resample('1h').sum().max()
        self.trader_workload['max_trades_per_day'] = self.data.data['info_offers_count'].resample('24h').sum().max()
        self.trader_workload['mean_trades_per_HH'] = self.data.data['info_offers_count'].mean()
        self.trader_workload['mean_trades_per_h'] =  self.data.data['info_offers_count'].resample('1h').sum().mean()
        self.trader_workload['mean_trades_per_d'] = self.data.data['info_offers_count'].resample('24h').sum().mean()
        self.trader_workload['periods_with_trades'] =  len(self.data.data[self.data.data['info_offers_count'] > 0])
        self.trader_workload['periods_without_trades'] =  len(self.data.data[self.data.data['info_offers_count'] == 0])
        self.trader_workload['trades_in_periods_with_trades'] = self.data.data[self.data.data['info_offers_count'] > 0]['info_offers_count'].mean()

        if not no_print:
            print('Maximum number of trades:\nPer HH: {:.0f}\nPer h: {:.0f}\nPer 24h: {:.0f}'.format(
                        self.trader_workload['max_trades_per_HH'],
                        self.trader_workload['max_trades_per_h'],
                        self.trader_workload['max_trades_per_day']))
            print('95% of all HH have {:.0f} trades or less'.format(self.trader_workload['95_trades_per_HH']))

            print('Mean number of trades:\nPer HH: {:.2f}\nPer h: {:.2f}\nPer 24h: {:.2f}\n'.format(
                                self.trader_workload['mean_trades_per_HH'],
                                self.trader_workload['mean_trades_per_h'],
                                self.trader_workload['mean_trades_per_d']))
            print('Number of periods with trades: {:.0f}, on average {:.2f} trades in those periods'.format(
                self.trader_workload['periods_with_trades'],
                self.trader_workload['trades_in_periods_with_trades']))

            print('Number of periods without trades: {:.0f},  so in {:.2f} of the periods there are trades.'.format(
                self.trader_workload['periods_without_trades'],
                self.trader_workload['periods_with_trades']/len(self.data.data)))

            print('\n')

        if detailed:
            if self.data.data['info_offers_count'].max() <= 1:
                x = [0, 1]
            else:
                x = range(1, self.data.data['info_offers_count'].max())
            y = [len(self.data.data[self.data.data['info_offers_count'] == i]['info_offers_count']) for i in x]

            plt.figure(figsize=(3, 4))
            plt.bar(left=x, height=y)
            plt.ylabel('Number of half hours')
            plt.xlabel('Number of trades')
            plt.show()



[docs]    def find_optimal_imbalance_settings(self, no_print = False, detailed=False):
        """Crude grid search to find the best window and margins
        """
        def find_val(m, w):
            self.schedules.set_imba_schedule('realistic', margin=m, window=w)
            self.contract.calc_imbalance_revenue()
            return (self.data.data['profit_imba'].sum()
                        /((self.data.data.index[-1] - self.data.data.index[0])
                            /pd.Timedelta('365day')),
                    self.data.data['profit_imba'].std())

        p = {}
        s = {}
        self.imbalance_settings = {}
        for m in np.linspace(0, 20, 21):
            for w in [1]: #range(1, 3) #In the interest of time saving, use only one window, it is always better than 2 windows
                p[m, w], s[m, w] = find_val(m, w)
        for key, value in p.items():
            if max(p.values()) == value:
                self.imbalance_settings['max_returns'] = key
                self.imbalance_settings['max_returns_std'] = s[key]


        if not no_print:
            print('Imbalance revenue is maximised if a window size of {} and a margin of {} is used.'.format(self.imbalance_settings['max_returns'][1],
                                                                                                             self.imbalance_settings['max_returns'][0]))
