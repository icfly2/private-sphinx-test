    """Data class 

    Param:
        refresh (bool):
        cut (bool):
        fishermen (list): 30, 85 

    Functions:
        schedules: Returns all schedule as (pd.DataFrame)
        da: Returns all da data as (pd.DataFrame)
        wd: Returns all within day data as (pd.DataFrame)
        imba: Returns allimbalance related data as (pd.DataFrame)
        el: Returns all electricity related as (pd.DataFrame)
        gas: Returns all gas as (pd.DataFrame)

    Attributes:
        wd_orders (pd.DataFrame): Apx order book
        timestep (str): Time step of the data, currently fixed at 30 minutes.
        data (pd.DataFrame): Main data frame
    """
    def __init__(self,asset,refresh=False, cut=True, fishermen=[30,85]):
        self._path = os.path.dirname(__file__)
        self.timestep = '30min'

        # wd orders
        self.wd_orders = pd.read_hdf(os.path.join(self._path, 'Intraday_data.h5'), 
                                                    key = 'APX_orderbook')
        # little clean up
        self.wd_orders = self.wd_orders[~self.wd_orders.index.duplicated(keep='first')]
        self.wd_orders['time_available'] = self.wd_orders['last_available']\
                                         - self.wd_orders['first_available']

        
        
        # remove fishermen offers
        if type(fishermen) == list:
            self.wd_orders = self.wd_orders[# Buy price below lowest ever cost
                                            ((self.wd_orders['Type'] == 'B') &
                                            (self.wd_orders['Price'] > fishermen[0])) |
                                            # Sell price below highest ever cost
                                            ((self.wd_orders['Type'] == 'S') &
                                            (self.wd_orders['Price'] < fishermen[1]))]

        if refresh:
            try:
                os.remove(os.path.join(self._path,
                                            '_' + asset.data.name + '_data_.pkl'))
            except FileNotFoundError:
                pass

        # Data
        try:
            self.data = pd.read_pickle(os.path.join(self._path,
                                        '_' + asset.data.name + '_data_.pkl'))
        except FileNotFoundError:
            self._update_asset_(asset,cut=cut)

    def _update_asset_(self,asset,cut=True):
        """Update the asset data.

        """
        da_prices = pd.read_pickle(os.path.join(self._path,'N2EX.pkl'))
        imba = pd.read_pickle(os.path.join(self._path,'imbalance.pkl'))
        embedded_cost = pd.read_csv(
                os.path.join(self._path,'embedded_cost.csv'),
                sep=';',index_col=['region','Time']
                ).loc[asset.data.location]

        embedded_cost.index = pd.to_datetime(
                                    embedded_cost.index,
                                    infer_datetime_format=True)

        embedded_cost['time_of_day'] = [ x.time() for x in embedded_cost.index]

        embedded_cost.index = embedded_cost.time_of_day

        gen_cost = pd.read_pickle(
            os.path.join(self._path,'gas_prices_from_prepp.pkl')
            )[['ask_price','WD_price']].resample('1h').ffill()/\
            asset.data.efficiency

        gen_cost['Time'] = gen_cost.index

        tqdm.pandas(desc='Loading import data')
        import_cost = gen_cost['Time'].progress_apply(
                   lambda x:
                   embedded_cost.loc[x.time()]['Mon-Fri_import']
                   if x.weekday() <= 4
                   else embedded_cost.loc[x.time()]['Sat-Sun_import'])
        tqdm.pandas(desc='Loading export data')                   
        export_cost = gen_cost['Time'].progress_apply(
                   lambda x:
                   embedded_cost.loc[x.time()]['Mon-Fri_export']
                   if x.weekday() <= 4
                   else embedded_cost.loc[x.time()]['Sat-Sun_export'])

        gen_cost['el_import']= import_cost
        gen_cost['el_export']= export_cost


        gen_cost.drop('Time',inplace=True,axis=1)

        # resample everything to timestep
        gen_cost = gen_cost.resample(self.timestep).ffill()

        # add Triad information
        gen_cost['Triad'] = pd.read_pickle(
                        os.path.join(self._path,'Triad_times.pkl'))

        gen_cost['CO2'] =   float(asset.data.CO_2_cost)
        gen_cost = gen_cost.resample('30min').ffill()
        da_prices = da_prices.resample('30min').ffill()

        # get all the orders from the orderbook and associate the ID's with
        # the HH they start in
        # Only using HH offers atm.
        d = {}
        HH_orders = self.wd_orders[self.wd_orders['Desc2'] == 'S_HH']
        for st in tqdm(HH_orders['start_time'].unique(),
                            desc='Collecting order data'):
            df = HH_orders[HH_orders['start_time'] == st].sort_values('first_available')

            d[st] = df.index.tolist()

        da_prices['Offers'] = pd.Series(d)

        
        if not cut:
            self.data = pd.concat([da_prices,imba,gen_cost],axis=1,join='outer')
        if cut:
            self.data = pd.concat([da_prices,imba,gen_cost],axis=1,join='inner')
            self.data.dropna(inplace=True)

        self.data.columns =['el_da', 'el_wd_offers', 'el_sip_imba', 'gas_da','gas_wd', 'el_import','el_export',  'el_triad', 'el_co2']

        # make empty output columnes
        self.data['schedule_da'] = ''
        self.data['schedule_wd'] = ''
        self.data['schedule_final_imba'] = ''
        self.data['schedule_contract'] = ''

        #self.data['info','da energy delivered'] = ''
        self.data['gas_heren'] = pd.read_pickle(
                    os.path.join(self._path,'gas_prices_from_prepp.pkl')
                    )['index_price'].resample('30min').ffill()/\
                    asset.data.efficiency
        self.data['info_offers_taken'] = ''
        self.data['payment_da'] = ''
        self.data['payment_deviation'] = ''
        self.data['payment_imba'] = ''

        self.data['profit_da'] = ''
        self.data['profit_wd'] = ''
        self.data['profit_imba'] = ''
        self.data['profit_total'] = ''

        self.data.to_pickle(os.path.join(self._path,
                                        '_' + asset.data.name + '_data_.pkl'))

    def get_column(self,col_value):
        cols = self.data.columns
        selection = []
        if type(col_value) == str:
            # col_value = list(col_value)
            for names in cols:
                if col_value in names.split('_'):
                    selection.append(True)
                else:
                    selection.append(False)
        elif type(col_value) == list:
            for names in cols:
                if bool([i for i in col_value if i in names.split('_')]):
                    selection.append(True)
                else:
                    selection.append(False)

        return self.data.loc[:,selection]

    def schedules(self):
        return self.get_column('schedule')
