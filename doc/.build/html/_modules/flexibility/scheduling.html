[docs]    def set_schedule(self, time, schedule):
        """Set the schedule for a time (DA, WD, Imbalance)

        The index must be in time step format. The schedule in MW
        of power operation. A schedule tha serves as an input to another
        operation is written to both dataframes. Should be used instead
        of the predefined methods if a custom schedule was computed.

        Args:
            time (str): Name of the market to schedule (DA,WD,Imbalance)
            schedule (pd.Series): Series of power settings. Must have the same
                                time index as the original schedule data frame.
                                Hourly data for DA, HH data for wd and imbalance.
        """
        if time[0] in 'dD':
            self.analysis.data.data['schedule_da'] = schedule
        elif time[0] in 'wW':
            self.analysis.data.data['schedule_wd'] = schedule
        elif time[0] in 'fFiI':
            self.analysis.data.data['schedule_final_imba'] = schedule
        elif time[0] in 'cC':
            self.analysis.data.data['schedule_contract'] = schedule




[docs]    def set_da_schedule(self, method='amp', **kwargs):
        """Automatically schedule the da operation.

        Schedule the DA operation using a predefined method. The follow methods are defined:

        - amp:
            - Initial scheduling method, according to which amp is reimbursed. Run when asset is 1 GBP in the money.
            - `kwargs`:
                - margin *float*: Margin by which the electricity price must exceed the gas prices, default = 1 GBP.
                - max_power *float*: Max power limit (%) of power to bid. Default is all of the assets power.
                - shift *float*: Number of hours in delay of price information.
                                For initial scheduling use 24, for contract use 0. Default is 0.
                - contract *bool*: Not for public use. If this is a contract calculation or not,
                    if contract, then the schedule is written to ['schedule','contract'].
                    The Da profit share for DONG is also calculated there.
        - battery:
            - Charge or discharge a battery in a fixed time interval. Requires the kwargs start_time, end_time and direction. 
                Always charges or discharges at the full asset power. Be careful, as this is an additive process and the schedule is
                altered, not replaced! Use method=lazy to initiate a clean slate.
            - `kwargs`:
                - start_time *string*: Time to start charging, eg: '05:00', make sure both the start end end times are
                        present in the index.
                - end_time *string*: End time of the interval in which to finish charging, eg: '06:00'. 
                - direction *string*: ('charge'/'discharge'), the direction of the flow in the battery.
        - lazy:
            - Do nothing. Asset is not scheduled on the day ahead market.

        Args:
            method (str): Method with which to schedule.
            **kwargs: Method dependant keyword arguments, see method description above for details.


        Example:
            For a battery to charge in the morning and discharge in the afternoon:

                analysis.schedules.set_da_schedule(method='lazy')
                analysis.schedules.set_da_schedule(method='battery',start_time = '05:00', end_time = '06:00', direction= 'charge')
                analysis.schedules.set_da_schedule(method='battery',start_time = '15:00', end_time = '16:00', direction= 'discharge')
        """
        if 'contract' not in kwargs:
            kwargs['contract'] = False

        if method == 'amp':
            if 'max_power' not in kwargs:
                kwargs['max_power'] = 1
            if 'margin' not in kwargs:
                kwargs['margin'] = 1
            if 'shift' not in kwargs:
                kwargs['shift'] = 0
            if 'market' not in kwargs:
                kwargs['market'] = 'heren'

            s = (self.analysis.data.data['el_da']-self.analysis.data.data['el_export'] > \
             (self.analysis.data.data['gas_'+kwargs['market']].shift(int(\
             kwargs['shift']/(self.analysis.data.timestep / pd.Timedelta('1h'))))\
                + self.analysis.data.data['el_co2']
                + kwargs['margin']))
            s = s*self.analysis.asset.power*kwargs['max_power']

        elif method == 'lazy':
            s = 0

        elif method == 'battery':
            if (('start_time' not in kwargs) or
                    ('end_time' not in kwargs) or
                    ('direction' not in kwargs)):
                raise TypeError('Please make sure you give appropriate start_time, end_time and direction arguments.')

            
            direction = self.analysis.asset.data.efficiency*(kwargs['direction']  == 'discharge')-(kwargs['direction']  == 'charge')

            s = ((self.analysis.data.data.index.time 
                >= pd.to_datetime(kwargs['start_time'],format='%H:%M').time())
                & 
                (self.analysis.data.data.index.time 
                < pd.to_datetime(kwargs['end_time'],format='%H:%M').time())
                )*self.analysis.asset.power*direction

            s += self.analysis.data.data['schedule_da']


        if kwargs['contract']:
            self.set_schedule('c',s)
        else:
            self.set_schedule('da',s)

        """Simulate the within day trade using a predefiend method

        The following trading methods are defined:

        - lazy:
            - Do not trade on the intraday
            - `kwargs`: None

        - greedy:
            - Take all profitable trades
            - `kwargs`:
                - capacity_charge (float): Charge per MWh with which to penalise deviations from the DA schedule.
                - margin (float): Pounds in total of profit that must be made per trade.
                - max_power_share (float): Proportion of max asset power to use for scheduling Default is 1, must be between 0 and 1.
                - max_trades (int): Maximum number of permissable trades per half hour trading block, default is 99.
                - trade_delay (pd.Timedelta): Time for which a trade must at least be available in the order book for it to be used in trading. 15 second granularity.

        - perfect:
            - Assuiming perfect market knowledge, select the best offers.
            - NOT IMPLIMENTED YET

        Args:
            method (str): Method with which to schedule.
            **kwargs: Method dependant keyword arguments, see method description above for details
        """
        # sett all those that are iterated and added to to start from zero
        self.analysis.data.data['profit_wd'] = 0
        self.analysis.data.data['payment_deviation'] = 0
        self.analysis.data.data['info_offers_count'] = 0
        self.analysis.data.data['info_offers_taken'] = ''
        
        if method == 'lazy':
            self.set_schedule('wd',self.analysis.data.data['schedule_da'])

        elif method == 'battery':
            self.analysis.asset.data.soc = 0
            self.analysis.data.data['info_wd_soc'] = 0
            if 'capacity' not in kwargs:
                raise TypeError('Must specify capacity of battery in MWh!')
            if 'max_trades' not in kwargs:
                kwargs['max_trades'] = 99
            if 'trade_delay' not in kwargs:
                kwargs['trade_delay'] = pd.Timedelta('0s')
            if 'stochastic' not in kwargs:
                kwargs['stochastic'] = False    
            if 'threshold' not in kwargs:
                raise TypeError('Need to set both buy and sell thresholds.')      
            if 'min_volume' not in kwargs:
                kwargs['min_volume'] = self.analysis.asset.power
            if 'max_cycles' not in kwargs:
                kwargs['max_cycles'] = 1
            df = self.analysis.data.data.head(100)
            time_factor = (self.analysis.data.timestep/pd.Timedelta('1h'))
            for row in df.itertuples():
            # eval trades as they come in
                profit_total = df.get_value(row.Index,'profit_wd')
                cur_power = 0#df.get_value(row.Index,'schedule_wd')
                da_set = df.get_value(row.Index,'schedule_da')
                offers_count = df.get_value(row.Index,'info_offers_count')
                offers_taken = df.get_value(row.Index,'info_offers_taken')
                ex_benefits = df.get_value(row.Index,'el_export')
                im_cost = df.get_value(row.Index,'el_import')

                if (df.loc[row.Index-pd.Timedelta('24h'):row.Index:,'info_wd_soc'].diff().apply(lambda x: abs(x)).sum()) > kwargs['max_cycles']:
                    offers_count = kwargs['max_trades']
                for offer_id in row.el_wd_offers:   
                    # Allow only max_trades per trading round
                    if offers_count >= kwargs['max_trades']:
                        break                                     
                    offer = self.analysis.data.wd_orders.loc[offer_id]
                    try:
                        if offer['time_available'] < kwargs['trade_delay']:
                            # don't evaluate the trade
                            pass
                        elif offer['Volume'] < kwargs['min_volume']:
                            # don't evaluate the trade
                            pass
                        elif (type(kwargs['stochastic']) == float)  & (kwargs['stochastic'] < np.random.random()):
                            # Randomly miss a deal
                            pass
                        elif (offer['Type'] == 'B') & (self.analysis.asset.data.soc > 0 ) & ((offer['Price']-ex_benefits) > kwargs['threshold']['sell']):
                            # the offer is taken, power is sold to the buy offer
                            cur_power = min(offer['Volume'],self.analysis.asset.power,self.analysis.asset.data.soc*kwargs['capacity']/0.5)
                            if self.analysis.asset.data.soc-cur_power*0.5/kwargs['capacity'] < -0.001:
                                print('WARNING!',self.analysis.asset.data.soc,cur_power)
                            soc_n = max(0,self.analysis.asset.data.soc-cur_power*0.5/kwargs['capacity'])
                            profit_total = (offer['Price']-ex_benefits)*(self.analysis.asset.data.soc-soc_n)*kwargs['capacity']*self.analysis.asset.data.efficiency                         
                            offers_count += 1
                            offers_taken += str(offer_id)+', '
                            self.analysis.asset.data.soc = soc_n
                        elif (offer['Type'] == 'S') & (self.analysis.asset.data.soc < 1 ) & ((offer['Price']+im_cost)< kwargs['threshold']['buy']):
                            cur_power = -min(offer['Volume'],self.analysis.asset.power,(1-self.analysis.asset.data.soc)*kwargs['capacity']/0.5)
                            if self.analysis.asset.data.soc+cur_power*0.5/kwargs['capacity'] > 1.0001:
                                print('WARNING!',self.analysis.asset.data.soc,cur_power)
                            soc_n = min(1,self.analysis.asset.data.soc-cur_power*0.5/kwargs['capacity'])                            
                            profit_total = (offer['Price']+im_cost)*(self.analysis.asset.data.soc-soc_n)*kwargs['capacity']
                            offers_count += 1
                            offers_taken += str(offer_id)+', '
                            self.analysis.asset.data.soc = soc_n
                    except:
                        import sys
                        errortype, value, tb = sys.exc_info()
                        pdb.set_trace()

                df = df.set_value(row.Index,'info_wd_soc',self.analysis.asset.data.soc)
                df = df.set_value(row.Index,'profit_wd',profit_total)
                df = df.set_value(row.Index,'schedule_wd', cur_power)
                df = df.set_value(row.Index,'schedule_da', da_set)
                df = df.set_value(row.Index,'info_offers_count', offers_count)
                df = df.set_value(row.Index,'info_offers_taken', offers_taken)
               

            self.analysis.data.data = df

        elif method == 'greedy':
            if 'capacity_charge' not in kwargs:

            if kwargs['parallel']:

            else:

                self.analysis.data.data = self.greedy(df,kwargs)

[docs]    def set_imba_schedule(self, method='realistic', **kwargs):
        """Automatically schedule the imbalance operation.

        The following scheduling methods are pre defined:

        - perfect:
            - Trigger imbalance when ever it is profitable to do so.
            - `kwargs`:
                - margin (float): Set the minimum profit what must be made in a half hour to trigger imbalance. Default is 0.

        - realistic:
            - Trigger imbalance when the worst of the preceeding imbalance cost periods, indicates the current period may be profitable.
            - `kwargs`:
                - margin (float): Additional profit requirment. Default is 0, can be negative.
                - window (int): Number of past half hour windows to consider. Default is 2.
                - stochastic (bool/float): Default is False, otherwise enter fraction of timesteps to scramble. Simulates wrong calls.

        -lazy:
            - Don't participate in the imbalance market.

        Args:
            method (str): Method with which to schedule.
            **kwargs: Method dependant keyword arguments, see method description above for details.

        """
        if method == 'perfect':
            kwargs = {}
            method = 'realistic'
            kwargs['shift_window'] = 0
            kwargs['window'] = 1
            kwargs['margin'] = 0
            
        if method == 'last_n_minutes':
            kwargs['window'] = 1
            kwargs['shift_window'] = 0
        
        if 'window' not in kwargs:
            kwargs['window'] = 1

        if 'margin' not in kwargs:
            kwargs['margin'] = 6

        if 'shift_window' not in kwargs:
            kwargs['shift_window'] = 1
        
        if 'decission_prices' not in kwargs:
            kwargs['decission_prices'] = self.analysis.data.data['el_sip_imba'].rolling(
                    window=kwargs['window'], min_periods=kwargs['window']
                    ).min().shift(kwargs['shift_window'])

        if 'charge_duration' not in kwargs:
            kwargs['charge_duration'] = 10 #minutes

        if method == 'lazy':
            final = self.analysis.data.data['schedule_wd']

        elif method == 'last_n_minutes':
            # TODO impliment Christians SIP analysis for Shaw
            self.analysis.asset.data.soc = 1
            final = []
            import pdb; pdb.set_trace()
            soc = []
            for p in kwargs['decission_prices']:
                if (p < kwargs['threshold']['buy']) & (self.analysis.asset.data.soc <= (60-kwargs['charge_duration'])/60 ):
                    self.analysis.asset.data.soc += kwargs['charge_duration']/60
                    final.append(-(kwargs['charge_duration']/30)*self.analysis.asset.power)
                elif(p > kwargs['threshold']['sell']) & (self.analysis.asset.data.soc >= (kwargs['charge_duration'])/60 ):
                    self.analysis.asset.data.soc -= kwargs['charge_duration']/60
                    final.append((kwargs['charge_duration']/30)*self.analysis.asset.power*self.analysis.asset.data['efficiency'])
                else:
                    final.append(0)
                soc.append(self.analysis.asset.data.soc)
            self.analysis.data.data['info_imba_soc'] = soc      

        elif method == 'realistic':
            # decissions
            capacity_up = float(self.analysis.asset.data.power) \
                     - self.analysis.data.data['schedule_wd']            
            s_up = (( (kwargs['decission_prices']
                     - self.analysis.data.data['el_export']
                     + kwargs['margin'])
                       > (self.analysis.data.data['gas_wd']
                       + self.analysis.data.data['el_co2']))
                        &  (capacity_up > 0 ))*capacity_up

            s_down = (( (kwargs['decission_prices']
                     - self.analysis.data.data['el_export']
                     + kwargs['margin'])
                       < (self.analysis.data.data['gas_wd']
                       + self.analysis.data.data['el_co2']))
                        # Check if capacity is available
                        &  (capacity_up < float(self.analysis.asset.data.power)))\
                        * (float(self.analysis.asset.data.power) - capacity_up)
            final =  self.analysis.data.data['schedule_wd']- s_down + s_up

            if 'stochastic' in kwargs:
                pdb.set_trace()
                scrambled_up = final.sample(frac=kwargs['stochastic'])
                scrambled_down = final.sample(frac=kwargs['stochastic'])
                final.loc[scrambled_up.index] = float(self.analysis.asset.data.power)
                final.loc[scrambled_down.index] = 0
        else:
            raise KeyError('Unknown method selected.')

        self.set_schedule('final',final)

[docs]    def greedy(self, df_chunk, kwargs):
        """

        note kwargs is a dictionary as greedy is called via functool.partial
        """
        time_factor = (self.analysis.data.timestep/pd.Timedelta('1h'))
        names = df_chunk.columns
        for row in df_chunk.itertuples():
            # eval trades as they come in
            profit_total = df_chunk.get_value(row.Index,'profit_wd')
            cur_power = df_chunk.get_value(row.Index,'schedule_wd')
            offers_count = df_chunk.get_value(row.Index,'info_offers_count')
            offers_taken = df_chunk.get_value(row.Index,'info_offers_taken')

            for offer_id in row.el_wd_offers:
                # Allow only max_trades per trading round
                if offers_count >= kwargs['max_trades']:
                    break

                offer = self.analysis.data.wd_orders.loc[offer_id]
                try:
                    if offer['time_available'] < kwargs['trade_delay']:
                        # don't evaluate the trade
                        pass
                    elif offer['Volume'] < kwargs['min_volume']:
                        # don't evaluate the trade
                        pass
                    elif (type(kwargs['stochastic']) == float)  & (kwargs['stochastic'] < np.random.random()):
                        # Randomly miss a deal
                        pass

                    elif offer['Type'] == 'B':
                        if cur_power < self.analysis.asset.power*kwargs['max_power_share']:
                            new_power = min(cur_power+offer['Volume'],
                                            self.analysis.asset.power*kwargs['max_power_share'])

                            change_in_power = new_power - cur_power  # +ve as buy offer


                            cap_charge = abs(new_power-row.schedule_contract)\
                                        *kwargs['capacity_charge']\

                            profit =(offer['Price'] #GBP/MWh
                                    -row.gas_wd
                                    -row.el_export
                                    -row.el_co2)

                            if profit*change_in_power - change_in_power*kwargs['margin'] - cap_charge*time_factor > 0:
                                profit_total += profit*time_factor*change_in_power
                                cur_power = new_power
                                offers_count += 1
                                offers_taken += str(offer_id)+', '

                    elif offer['Type'] == 'S':
                        if cur_power > 0:# feasible
                            #profitable

                            new_power = max(df_chunk.get_value(row.Index,'schedule_wd') - offer['Volume'],0)
                            change_in_power = cur_power -  new_power  # +ve as sell offer
                            cap_charge = abs(new_power-row.schedule_contract)\
                                        *kwargs['capacity_charge']\
                                        *time_factor

                            profit =((row.gas_wd #GBP/MWh
                                    +row.el_export
                                    +row.el_co2)#GBP/MWh for a half hour)
                                    -offer['Price'])*time_factor# Energy is only sold for a half hour

                            if profit*change_in_power - change_in_power*kwargs['margin'] - cap_charge*time_factor > 0:
                                profit_total += profit*time_factor*change_in_power
                                cur_power = new_power
                                offers_count += 1
                                offers_taken += str(offer_id)+', '
                    else:
                        raise KeyError('Unknown offer type encountered', offer['Type'])


                except ValueError:
                    pdb.set_trace()

            df_chunk = df_chunk.set_value(row.Index,'profit_wd',profit_total)
            df_chunk = df_chunk.set_value(row.Index,'schedule_wd', cur_power)
            df_chunk = df_chunk.set_value(row.Index,'info_offers_count', offers_count)
            df_chunk = df_chunk.set_value(row.Index,'info_offers_taken',offers_taken)

        df_chunk.columns = names

        return df_chunk


    def parallelize_dataframe(self, df, func, kwargs):
        num_cores = multiprocessing.cpu_count()-1  #number of cores on your machine (leave one free to not freeze machine
        num_partitions = num_cores #number of partitions to split dataframe
        df_split = np.array_split(df, num_partitions)
        pool = multiprocessing.Pool(num_cores)
        result  = pool.map(partial(func, kwargs=kwargs), df_split)
        df = pd.concat(result)

        pool.close()
        pool.join()
        return df
