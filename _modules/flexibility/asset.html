[docs]class Asset(object):
    """Define an energy asset.

    Optionally set new to the name of a predefined asset to define
    the asset on initialisation.

    Attributes:
        all_assets: list of all already defined assets
        all_parameters: list of all already defiend parameters
        units: dictionary describing the parameter units

    Args:
        new (string): Name of the new asset, which must already excisit in
        the database. Providing this name calls set_asset.

    Examples:
            Create and set an asset as amp:

            >>> import flexibility as fl
            >>> A = fl.asset.Asset('amp')
    """
    def __init__(self, new=None):
        self.assets_data = 'H:\\flexibility\\flexibility\\data\\assets.csv'
        df = pd.read_csv(self.assets_data, sep=';',index_col=0)
        self.all_assets = list(df.index[1:]) # excude the unit line
        self.units = dict(df.loc['unit'].dropna())
        self.all_parameters = list(df.columns)

        # for internal use only
        self._required_keys = ["power", 'location','CO_2_cost','latest_data']
        if new:
            self.set_asset(new)


[docs]    def set_asset(self, name):
        """Define the asset.

        Loads a predefiend asset from the file assets.csv
        Defines the asset with these attributes:

        Args:
            power (float): Power of the asset in MW.
            desc (string): Long text description of the asset.
            data (pd.Series): The entire entry of parameters defining the asset.

        Examples:
            Create and set an asset as amp:

            >>> import flexibility as fl
            >>> A = fl.asset.Asset()
            >>> A.set_asset('amp')


        """
        self.data = pd.read_csv(self.assets_data, sep=';', index_col=0).loc[name]
        self.power = float(self.data.power)
        self.desc = self.data.description



[docs]    def add_asset(self, name, data):
        """Add a new asset by writing the information to assets.csv

        Be sure to enter information in the correct units, and not use
        semi-colons in the asset description text. Also all required data tags
        must be supplied, else a KeyError will be raised, stating the missing
        keys.

        Args:
            name (string): Name of the new asset, must be unique as it is
            used as index in the database.
            data (dict):The data and parameters of the asset.

        Returns:
            None

        Raises:
            KeyError: If required data wasn't specified or an existing key was
                     used to name a new asset.

        Examples:
            Add an asset called *example* with these parameters:
            - Power: 10 MW
            - Description: This is an example asset.
            - Location: UK_SE
            - Efficiency: 0.33


            >>> import flexibility as fl
            >>> A = fl.asset.Asset()
            >>> data ={
                    'power':10,
                    'description':'This is an example asset',
                    'efficiency':0.33,
                    'location':'UK_SE',
                    'realised':False,
                    }
            >>> A.add_asset(name='example',data=data)
            >>> A.all_assets #check the new asset is listed.
        """
        # load all asset data
        df = pd.read_csv(self.assets_data, sep=';', index_col=0)

        if name not in df.index:
            # key is unique, check for all required data:
            missing_key = set(self._required_keys) -set(data.keys())
            if not missing_key:
                # All required keys are present
                new_asset = pd.Series(data=data, name=name)
                df = df.append(new_asset)

                # rewrite csv
                with open(self.assets_data, 'w') as f:
                    df.to_csv(f, header=True, sep=';')
            else:
                # Raise an error
                raise KeyError("A required key: "+ " ".join(missing_key) +" was not specified!")

        else:
            raise KeyError('The asset '+name+"is already defined. Check attribute Asset.all_assets for a list of defined assets. Use modify_asset to alter an excisiting entry")



[docs]    def modify_asset(self, name, data):
        """Modify the information for an excisting asset

        Replaces the excising data in the assets.csv file with the new data.
        Prints the changes to screen if verbose=True

        Args:
            name: string, setting the name of the asset
            data: dictionary of data to be modified
        """
         # load all asset data
        df = pd.read_csv(self.assets_data, sep=';', index_col=0)

        if name in df.index:
            # key is already present:
            for col, val in data.items():
                df.set_value(index=name, col=col, value=val)

        else:
            raise KeyError('The key ' + name + 
                ' is not yet defined. Check Asset.all_assets for a list of defined assets. use add_asset to add a new asset to the database.')
