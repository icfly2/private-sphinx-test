[docs]    def calc_da_contract_obligations(self, contract='amp', list_of_contracts=False,**kwargs):
        """Calculate the run schedule and payments to customer
        Based on a contract and certain market data, gas and N2EX prices
        in the case of AMP, the customer is entitled to a certain payment.
        This payment is calculated by assuming a certain engine operating
        regime. This regime serves to calculate the operating profits and
        capacity payments.
        Contracts:
        - amp:
            - Run if operation appears profitable with a margin of 1 GBP based on the heren DA index price and the N2EX price.
            - 90% of the resulting profit are passed on to the customer, 10%
                are retained by DONG. The actual profit for DONG is given by the actual day ahead operation minus the share given to AMP

        - shaw:
            - Trade power on the N2EX as governed by the da schedule.
            - All profits are kept.

        Args:
            contract (str): The type of contract.
            list_of_contracts (bool): If true does not calculate anything but
                prints a list of contracts encoded.
            kwargs:
                - stochastic (pd.Series) or (float): With the index of self.data.data
                    this pd.Series is added to the profit of da operations.
                    Example: So a 1.96 std increase of the da cost is:
                        -1.96*self.data.data['gas_da'].std()


        """
        contracts = ['amp', 'shaw']
        if list_of_contracts == True:
            print('Contracts encoded: ',contracts)
            return None

        if contract == 'amp':
            self.analysis.schedules.set_da_schedule('amp',contract=True, market = 'da')

            #if type(self.analysis.data.data['schedule', 'final'][0]) is not str:
            #    print('Remember to update the capacity charge')

            # GBP/MWH
            profit = self.analysis.data.data['el_da']\
                    -self.analysis.data.data['gas_da']\
                    -self.analysis.data.data['el_export']\
                    -self.analysis.data.data['el_co2']

            money_contract = profit\
                   *self.analysis.data.data['schedule_contract']\
                   *(self.analysis.data.timestep / pd.Timedelta('1h'))

            if 'stochastic' in kwargs:
                profit += kwargs['stochastic']


            money_real = profit\
                   *self.analysis.data.data['schedule_da']\
                   *(self.analysis.data.timestep / pd.Timedelta('1h'))

            # remove -0.000, error is raised if actual negatives excists
            if (money_contract < -0.01).any():
                #pdb.set_trace()
                raise ValueError('Something went wrong in the profit calculations')
            money_contract = abs(money_contract)

            self.analysis.data.data['payment_da'] = 0.9*money_contract
            self.analysis.data.data['profit_da'] = money_real-0.9*money_contract

        elif contract == 'shaw':
            # Just simple difference between charging and discharging times and prices
            # export_rev = (self.analysis.data.data['reward', 'da']-self.analysis.data.data['cost','da']*(self.analysis.data.data['schedule', 'da'] > 0)
            # import_rev
            self.analysis.data.data['profit_da'] = self.analysis.data.data['schedule_da']\
            *(self.analysis.data.data['el_da']
            -self.analysis.data.data['el_export']*(self.analysis.data.data['schedule_da'] > 0)
            +self.analysis.data.data['el_import']*(self.analysis.data.data['schedule_da'] < 0)
            +0*(self.analysis.data.data['schedule_da'] == 0))\
            *(self.analysis.data.timestep / pd.Timedelta('1h'))
            
            self.analysis.data.data['payment_da'] = 0

 

[docs]    def calc_capacity_payment(self,charge=5, monthly=False):
        """Accuratly calculate the capacity payment.

        Calculate the capacity payment, considering monthly
        minimum requirements as well as overflows from previous months.

        Args:
            charge (float): The charge for devation from the da schedule
            monthly (bool): Is the capacity payment settled on a monthly bases.
                Default is False. NOT IMPLIMENTED YET

        """
        deviation = abs(
            self.analysis.data.data['schedule_contract']
            -self.analysis.data.data['schedule_wd']
            )

        self.analysis.data.data['payment_deviation'] = deviation*charge




[docs]    def calc_imbalance_revenue(self,method='amp',rev_split=0.5,equal_split=False,run_time=10):
        """Calculate and share the imbalance revenue

        Profits are devided according to the split, losses covered by DONG:

        Args:
            method (str): Name of the method used to compute the imbalance revenue
                - 'amp': for a  generator where imbalance is the deviation from the within day schedule. With generation cost.
                - 'battery': For battery or other asset, buying and selling only on the SIP, without generation cost. 
            rev_split (float): Share of imbalance revenue retained
                        by dong, must be between 0 and 1, defaults to 0.5.
        """
        
        #MW
        change_for_imba = self.analysis.data.data['schedule_final_imba']\
                        -self.analysis.data.data['schedule_wd']
        if method == 'amp':
            #GBP
            #import pdb; pdb.set_trace()

            imba_revenue = (self.analysis.data.data['el_sip_imba'] +
                            self.analysis.data.data['el_co2'] 
                          +np.sign( self.analysis.data.data['gas_wd']
                           - (self.analysis.data.data['el_sip_imba']
                            + self.analysis.data.data['el_export']
                             + self.analysis.data.data['el_co2'])
                             )*self.analysis.data.data['el_export'] 
                            - self.analysis.data.data['gas_wd'])\
                             * change_for_imba\
                             * (self.analysis.data.timestep/pd.Timedelta('1h'))

            # imba_revenue = (self.analysis.data.data['el_sip_imba'] 
            #               - self.analysis.data.data['el_co2'] 
            #               - self.analysis.data.data['gas_wd']
            #               - self.analysis.data.data['el_export']
            #               - self.analysis.data.data['gas_wd'])\
            #                  * change_for_imba\
            #                  * (self.analysis.data.timestep/pd.Timedelta('1h'))

        elif method == 'battery':
            imba_revenue = change_for_imba*\
            (self.analysis.data.data['el_sip_imba']\
            +(self.analysis.data.data['schedule_final_imba'] > 0)*self.analysis.data.data['el_export']
            -(self.analysis.data.data['schedule_final_imba'] < 0)*self.analysis.data.data['im_export'])\
            *(run_time/pd.Timedelta('60min'))


        # Profitable only is shared:
        if not equal_split:
            self.analysis.data.data['payment_imba'] = (imba_revenue*(1-rev_split))\
                                                            *(imba_revenue > 0)
            self.analysis.data.data['profit_imba'] = imba_revenue*rev_split*(imba_revenue > 0)\
                                                        + imba_revenue*(imba_revenue <= 0)

        else:
            self.analysis.data.data['payment_imba'] = imba_revenue*(1-rev_split)
            self.analysis.data.data['profit_imba'] = imba_revenue*rev_split
